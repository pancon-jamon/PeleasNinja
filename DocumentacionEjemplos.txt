def suma(a, b):
    """
    Suma dos números y devuelve el resultado.
    
    Args:
        a (float/int): Primer número.
        b (float/int): Segundo número.
    
    Returns:
        float/int: Resultado de la suma.
    
    Ejemplo:
        >>> suma(3, 5)
        8
    """
    return a + b
